create table Ciudad(
	Ciudad_Id  int,
	Ciudad_Detalle varchar(50) not null,
	Ciudad_Obs varchar (200) not null default 'sin observaciones',
	primary key (Ciudad_Id),
	);
create table Actividad(
	Actividad_Id int,
	Actividad_Detall varchar(50) not null,
	Actividad_Obs varchar (200) not null default 'sin observaciones',
	primary key (Actividad_Id),
	);

create table Persona(
	Rut_Persona varchar(8),
	Digito_Verificador char not null,
	Edad_Persona int not null,
	Nombre_Persona varchar (30) not null,
	Apellido_Pater_Persona varchar(15) not null,
	Apellido_Mater_Persona varchar(15) not null,
	primary key(Rut_Persona),
	Ciudad_Id int,
	Actividad_Id int,
	constraint fk_CiudadId foreign key (Ciudad_Id)references Ciudad(Ciudad_Id),
	constraint fk_ActividadId foreign key (Actividad_Id)references Actividad(Actividad_Id), 
    );
